# 智能电网光伏负荷预测系统方案

## 一、项目概述

### 1.1 系统目标
构建一个基于实时气象数据的智能电网光伏发电负荷预测系统，实现：
- **实时性**：通过Open-Meteo API获取美国新英格兰地区实时天气数据
- **精准预测**：利用已训练的深度学习模型预测未来24小时电力负荷
- **智能化**：结合光伏发电特性，优化电网调度策略
- **可扩展**：支持多区域、多时间尺度预测

### 1.2 技术架构
```
[Open-Meteo API] → [数据采集模块] → [特征工程模块] → [深度学习模型] → [预测结果]
       ↓                                                              ↓
[美国新英格兰地区]                                           [可视化面板 + API接口]
```

#### 2.1.2 数据映射与转换
**Open-Meteo → 模型输入特征映射**：

| Open-Meteo参数 | 单位 | 训练数据特征 | 转换规则 |
|---------------|------|-------------|----------|
| temperature_2m | °C | Dry_Bulb | 直接使用 |
| dew_point_2m | °C | Dew_Point | 直接使用 |
| relative_humidity_2m | % | humidity_index | 通过温湿度计算 |
| wind_speed_10m | m/s | wind_index | 单位转换 |
| cloud_cover | % | cloud_index | 直接使用 |
| shortwave_radiation | W/m² | solar_index | 光伏发电相关性 |

**单位转换**：
```python
def convert_weather_data(api_data):
    """将Open-Meteo数据转换为模型输入格式"""
    
    # 1. 温度直接使用（Open-Meteo返回°C，训练数据也是°C）
    dry_bulb = api_data['temperature_2m']
    dew_point = api_data['dew_point_2m']
    
    # 2. 湿度指标计算（与训练数据保持一致）
    humidity_index = dry_bulb - dew_point
    
    # 3. 风速单位转换 m/s → mph
    wind_speed_mph = api_data['wind_speed_10m'] * 2.23694
    
    # 4. 云量处理
    cloud_cover = api_data['cloud_cover']
    
    # 5. 太阳辐射特征（新增光伏发电相关特征）
    solar_radiation = api_data.get('shortwave_radiation', 0)
    
    return {
        'Dry_Bulb': dry_bulb,
        'Dew_Point': dew_point, 
        'humidity_index': humidity_index,
        'wind_speed': wind_speed_mph,
        'cloud_cover': cloud_cover,
        'solar_radiation': solar_radiation
    }
```

### 2.2 特征工程适配模块

#### 2.2.1 滞后特征生成
```python
class RealTimeFeatureGenerator:
    """实时特征生成器"""
    
    def __init__(self, historical_data_path):
        # 加载历史数据用于滞后特征计算
        self.historical_df = pd.read_pickle(historical_data_path)
        self.latest_timestamp = None
        
    def generate_lag_features(self, current_data):
        """生成滞后特征"""
        # 合并当前数据到历史数据末尾
        temp_df = pd.concat([self.historical_df, current_data], axis=0)
        
        # 生成滞后特征（与训练阶段完全一致的逻辑）
        temp_df = self._add_load_lags(temp_df)
        temp_df = self._add_temp_lags(temp_df)
        
        # 返回最新时刻的滞后特征
        latest_features = temp_df.iloc[-1:]
        return latest_features
        
    def _add_load_lags(self, df):
        """负荷滞后特征"""
        df['load_lag_1h'] = df['System_Load'].shift(1)
        df['load_lag_24h'] = df['System_Load'].shift(24)
        df['load_lag_48h'] = df['System_Load'].shift(48)
        df['load_lag_168h'] = df['System_Load'].shift(168)
        return df
        
    def _add_temp_lags(self, df):
        """温度滞后特征"""
        df['temp_lag_24h'] = df['Dry_Bulb'].shift(24)
        df['temp_lag_168h'] = df['Dry_Bulb'].shift(168)
        return df
```

```python
def generate_rolling_features(df):
    """生成滚动窗口特征"""
    
    # 负荷滚动特征
    df['load_rolling_mean_24h'] = df['System_Load'].shift(1).rolling(24).mean()
    df['load_rolling_std_24h'] = df['System_Load'].shift(1).rolling(24).std()
    df['load_rolling_mean_168h'] = df['System_Load'].shift(1).rolling(168).mean()
    
    # 温度滚动特征
    df['temp_rolling_mean_24h'] = df['Dry_Bulb'].shift(1).rolling(24).mean()
    df['temp_rolling_max_24h'] = df['Dry_Bulb'].shift(1).rolling(24).max()
    df['temp_rolling_min_24h'] = df['Dry_Bulb'].shift(1).rolling(24).min()
    
    return df
```

#### 2.2.3 时间特征生成
```python
def generate_time_features(df):
    """生成时间特征（与训练阶段完全一致）"""
    
    # 提取时间特征
    df['hour'] = df.index.hour
    df['day_of_week'] = df.index.dayofweek
    df['day_of_year'] = df.index.dayofyear
    df['month'] = df.index.month
    
    # 正余弦编码（与训练阶段相同的逻辑）
    df['hour_sin'] = np.sin(2 * np.pi * df['hour'] / 24)
    df['hour_cos'] = np.cos(2 * np.pi * df['hour'] / 24)
    df['doy_sin'] = np.sin(2 * np.pi * df['day_of_year'] / 365.25)
    df['doy_cos'] = np.cos(2 * np.pi * df['day_of_year'] / 365.25)
    
    # 节假日特征
    df['is_weekend'] = df['day_of_week'].isin([5,6])
    df['is_holiday'] = df.index.isin(get_holiday_calendar('US_NewEngland'))
    
    return df
```

#### 2.3.1 实时预测服务
```python
class RealTimePredictionService:
    """实时预测服务"""
    
    def __init__(self, models_path):
        # 加载训练好的模型
        self.models = self._load_models(models_path)
        self.scaler = self._load_scaler()
        
    def predict_realtime(self, weather_data):
        """实时预测主函数"""
        
        # 1. 特征工程
        features = self._feature_engineering(weather_data)
        
        # 2. 数据归一化
        features_scaled = self.scaler.transform(features)
        
        # 3. 序列化（lookback=168）
        X = self._create_sequences(features_scaled, lookback=168)
        
        # 4. 多模型预测
        predictions = {}
        for model_name, model in self.models.items():
            pred = model.predict(X, verbose=0)
            predictions[model_name] = pred
            
        # 5. 集成预测结果
        ensemble_pred = self._ensemble_predictions(predictions)
        
        return ensemble_pred
        
    def _ensemble_predictions(self, predictions):
        """集成预测结果（使用训练阶段确定的最优权重）"""
        # 权重来自训练阶段验证集表现
        weights = {
            'EnhancedLSTM': 0.35,
            'BiGRU': 0.30, 
            'DeepTCN': 0.20,
            'SpatialTransformer': 0.15
        }
        
        ensemble_result = np.zeros_like(predictions['EnhancedLSTM'])
        for model_name, pred in predictions.items():
            ensemble_result += weights[model_name] * pred
            
        return ensemble_result
```

### 2.4 光伏发电集成模块

#### 2.4.1 光伏发电量估算
```python
def estimate_pv_generation(solar_data, weather_conditions):
    """基于天气条件和历史数据估算光伏发电量"""
    
    # 光伏出力计算公式
    def pv_efficiency_model(radiation, temperature, cloud_cover):
        """光伏发电效率模型"""
        
        # 基础转换效率（晶硅组件典型值）
        base_efficiency = 0.18
        
        # 温度损失系数
        temp_coefficient = -0.004  # %/°C
        temp_loss = temp_coefficient * (temperature - 25)
        
        # 云量遮挡损失
        cloud_loss = 0.8 * (cloud_cover / 100)
        
        # 综合效率
        actual_efficiency = base_efficiency * (1 + temp_loss) * (1 - cloud_loss)
        
        # 计算发电功率（kW）
        pv_power = radiation * actual_efficiency * installed_capacity
        
        return pv_power
    
    # 新英格兰地区装机总容量估算
    installed_capacity = 2500  # MW (示例数据)
    
    # 计算每小时发电量
    hourly_generation = []
    for hour_data in solar_data:
        power = pv_efficiency_model(
            hour_data['radiation'],
            hour_data['temperature'],
            hour_data['cloud_cover']
        )
        hourly_generation.append(power)
        
    return hourly_generation
```

#### 2.4.2 净负荷计算

```python
def calculate_net_load(total_load_prediction, pv_generation):
    """计算净负荷（总负荷 - 光伏发电）"""
    
    net_load = []
    for i in range(len(total_load_prediction)):
        # 预测时段为白天时考虑光伏发电
        if 6 <= i <= 18:  # 小时范围
            net = total_load_prediction[i] - pv_generation[i]
        else:
            net = total_load_prediction[i]  # 夜间无光伏发电
            
        net_load.append(max(0, net))  # 确保非负
        
    return net_load
```

### 4.1 异常值处理
```python
class DataQualityMonitor:
    """数据质量监控器"""
    
    def validate_weather_data(self, data):
        """验证气象数据质量"""
        
        # 温度范围检查（新英格兰地区合理范围）
        if not (-30 <= data['temperature'] <= 45):
            raise ValueError(f"温度异常: {data['temperature']}°C")
            
        # 露点检查（露点不能高于气温）
        if data['dew_point'] > data['temperature']:
            logger.warning(f"露点{data['dew_point']}高于气温{data['temperature']}")
            
        # 云量范围检查
        if not (0 <= data['cloud_cover'] <= 100):
            raise ValueError(f"云量异常: {data['cloud_cover']}%")
            
    def handle_missing_data(self, data):
        """处理缺失数据"""
        
        # 使用历史同期数据插值
        if pd.isna(data['temperature']):
            data['temperature'] = self._get_historical_similar_hour('temperature')
            
        # 线性插值适用于短时间缺失
        if pd.isna(data['dew_point']):
            data['dew_point'] = self._interpolate_dew_point()
            
        return data
```
